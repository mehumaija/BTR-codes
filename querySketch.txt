  	UNION {
	  	?evidence :reference ?reference.
  		?reference :from ?doi.
	  	FILTER REGEX(STR(?doi), "DOI:")

	{ SELECT DISTINCT ?protein1 ?protein2 ?interaction

			?iso1 :interaction ?interaction.

			    {?protein1 :isoform ?iso1.

	}
